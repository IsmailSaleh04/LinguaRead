-- Enable UUID extension
CREATE EXTENSION IF NOT EXISTS "uuid-ossp";

-- Profiles table
CREATE TABLE profiles (
  id UUID PRIMARY KEY REFERENCES auth.users(id) ON DELETE CASCADE,
  email TEXT,
  native_language TEXT,
  target_language TEXT,
  proficiency_level TEXT,
  interests TEXT[],
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- Articles table
CREATE TABLE articles (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  title TEXT NOT NULL,
  summary TEXT,
  cached_text TEXT NOT NULL,
  url TEXT,
  topic TEXT,
  word_count INTEGER,
  difficulty TEXT,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- User articles (junction table for user-article relationship)
CREATE TABLE user_articles (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  user_id UUID REFERENCES auth.users(id) ON DELETE CASCADE,
  article_id UUID REFERENCES articles(id) ON DELETE CASCADE,
  progress INTEGER DEFAULT 0,
  completed BOOLEAN DEFAULT FALSE,
  started_at TIMESTAMPTZ DEFAULT NOW(),
  completed_at TIMESTAMPTZ,
  UNIQUE(user_id, article_id)
);

-- User words table
CREATE TABLE user_words (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  user_id UUID REFERENCES auth.users(id) ON DELETE CASCADE,
  word TEXT NOT NULL,
  status TEXT NOT NULL CHECK (status IN ('unknown', 'learning', 'known')),
  difficulty TEXT,
  definition TEXT,
  context TEXT,
  last_seen TIMESTAMPTZ DEFAULT NOW(),
  times_seen INTEGER DEFAULT 1,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  UNIQUE(user_id, word)
);

-- Click analytics table
CREATE TABLE click_analytics (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  user_id UUID REFERENCES auth.users(id) ON DELETE CASCADE,
  article_id UUID REFERENCES articles(id) ON DELETE CASCADE,
  word TEXT NOT NULL,
  clicked_at TIMESTAMPTZ DEFAULT NOW()
);

-- Indexes for better query performance
CREATE INDEX idx_user_articles_user_id ON user_articles(user_id);
CREATE INDEX idx_user_articles_article_id ON user_articles(article_id);
CREATE INDEX idx_user_words_user_id ON user_words(user_id);
CREATE INDEX idx_user_words_status ON user_words(status);
CREATE INDEX idx_click_analytics_user_id ON click_analytics(user_id);
CREATE INDEX idx_click_analytics_word ON click_analytics(word);

-- Row Level Security (RLS) policies
ALTER TABLE profiles ENABLE ROW LEVEL SECURITY;
ALTER TABLE user_articles ENABLE ROW LEVEL SECURITY;
ALTER TABLE user_words ENABLE ROW LEVEL SECURITY;
ALTER TABLE click_analytics ENABLE ROW LEVEL SECURITY;
ALTER TABLE articles ENABLE ROW LEVEL SECURITY;

-- Profiles policies
CREATE POLICY "Users can view their own profile"
  ON profiles FOR SELECT
  USING (auth.uid() = id);

CREATE POLICY "Users can update their own profile"
  ON profiles FOR UPDATE
  USING (auth.uid() = id);

CREATE POLICY "Users can insert their own profile"
  ON profiles FOR INSERT
  WITH CHECK (auth.uid() = id);

-- Articles policies (public read)
CREATE POLICY "Anyone can view articles"
  ON articles FOR SELECT
  TO authenticated
  USING (true);

CREATE POLICY "Anyone can create articles"
  ON articles FOR INSERT
  TO authenticated
  WITH CHECK (true);

-- User articles policies
CREATE POLICY "Users can view their own articles"
  ON user_articles FOR SELECT
  USING (auth.uid() = user_id);

CREATE POLICY "Users can create their own article relationships"
  ON user_articles FOR INSERT
  WITH CHECK (auth.uid() = user_id);

CREATE POLICY "Users can update their own article relationships"
  ON user_articles FOR UPDATE
  USING (auth.uid() = user_id);

-- User words policies
CREATE POLICY "Users can view their own words"
  ON user_words FOR SELECT
  USING (auth.uid() = user_id);

CREATE POLICY "Users can create their own words"
  ON user_words FOR INSERT
  WITH CHECK (auth.uid() = user_id);

CREATE POLICY "Users can update their own words"
  ON user_words FOR UPDATE
  USING (auth.uid() = user_id);

-- Click analytics policies
CREATE POLICY "Users can view their own analytics"
  ON click_analytics FOR SELECT
  USING (auth.uid() = user_id);

CREATE POLICY "Users can create their own analytics"
  ON click_analytics FOR INSERT
  WITH CHECK (auth.uid() = user_id);

-- Function to automatically create profile on signup
CREATE OR REPLACE FUNCTION public.handle_new_user()
RETURNS TRIGGER AS $
BEGIN
  INSERT INTO public.profiles (id, email, created_at, updated_at)
  VALUES (NEW.id, NEW.email, NOW(), NOW());
  RETURN NEW;
END;
$ LANGUAGE plpgsql SECURITY DEFINER;

-- Trigger to call the function
CREATE TRIGGER on_auth_user_created
  AFTER INSERT ON auth.users
  FOR EACH ROW EXECUTE FUNCTION public.handle_new_user();

# Langpilto - Language Learning Platform

## Setup Instructions

### 1. Install Dependencies
```bash
npm install
```

### 2. Supabase Setup
1. Create a new Supabase project at https://supabase.com
2. Go to SQL Editor and run the migration script from `supabase/migrations/001_initial_schema.sql`
3. Copy your project URL and anon key from Settings > API
4. Create `.env.local` file and add your credentials:
```
NEXT_PUBLIC_SUPABASE_URL=your_project_url
NEXT_PUBLIC_SUPABASE_ANON_KEY=your_anon_key
```

### 3. Run Development Server
```bash
npm run dev
```

Visit http://localhost:3000

### 4. Features Overview

**Authentication:**
- Register/Login with email & password
- Automatic profile creation on signup

**Onboarding:**
- Set native and target languages
- Choose proficiency level (A1-C2)
- Select interests for personalized recommendations

**Dashboard:**
- View learning statistics
- Search Wikipedia for articles
- Continue reading saved articles

**Article Reading:**
- Click any word for instant definition
- Mark words as Known/Learning/Unknown
- Color-coded vocabulary tracking
- Track reading progress

**Word Bank:**
- View all encountered words
- Filter by status (Known/Learning/Unknown)
- Search and track vocabulary
- See usage statistics

**Profile:**
- Update languages and proficiency
- Manage interests
- View account information

### 5. Database Tables

- **profiles**: User learning preferences
- **articles**: Wikipedia articles cache
- **user_articles**: User reading progress
- **user_words**: Vocabulary tracking
- **click_analytics**: Word interaction analytics

### 6. Key Features

✅ Wikipedia article integration
✅ Real-time word definitions
✅ Vocabulary tracking system
✅ Progress monitoring
✅ Click analytics
✅ Dark/Light theme
✅ Responsive design
✅ RLS security policies

### 7. API Routes

- `/api/profile/*` - User profile management
- `/api/articles/*` - Article operations
- `/api/words/*` - Vocabulary tracking
- `/api/wiki/*` - Wikipedia integration

### Tech Stack

- **Frontend**: Next.js 14, React, TypeScript, Tailwind CSS
- **Backend**: Next.js API Routes
- **Database**: Supabase (PostgreSQL)
- **Authentication**: Supabase Auth
- **APIs**: Wikipedia API, Free Dictionary API

### Troubleshooting

**Issue:** Supabase connection fails
- Check your .env.local file has correct credentials
- Verify Supabase project is active

**Issue:** RLS policies blocking queries
- Ensure you're authenticated
- Check the SQL migration ran successfully

**Issue:** Word definitions not loading
- The Free Dictionary API may be rate-limited
- Words will show "Definition not available" if lookup fails

For support, check the Supabase dashboard logs and browser console.

"use client";

import { Menu, Sun, Moon, User, LogOut } from "lucide-react";
import Link from "next/link";
import { useState, useRef, useEffect } from "react";
import { supabase } from "@/lib/supabase";
import { useRouter } from "next/navigation";

export default function Navbar({ onToggleSidebar, onToggleTheme, currentTheme }) {
  const router = useRouter();
  const [profileOpen, setProfileOpen] = useState(false);
  const [user, setUser] = useState<any>(null);
  const dropdownRef = useRef<HTMLDivElement>(null);

  useEffect(() => {
    loadUser();

    const handleClickOutside = (event: MouseEvent) => {
      if (dropdownRef.current && !dropdownRef.current.contains(event.target as Node)) {
        setProfileOpen(false);
      }
    };

    document.addEventListener("mousedown", handleClickOutside);
    return () => document.removeEventListener("mousedown", handleClickOutside);
  }, []);

  async function loadUser() {
    const { data: { user } } = await supabase.auth.getUser();
    setUser(user);
  }

  async function handleLogout() {
    await supabase.auth.signOut();
    router.push("/auth/login");
  }

  return (
    <header className="h-16 bg-[var(--bg-secondary)] border-b border-[var(--border-primary)] flex items-center justify-between px-6 shadow-sm">
      <div className="flex items-center gap-4">
        <button
          className="md:hidden p-2 text-[var(--text-secondary)] hover:text-[var(--accent-primary)] transition"
          onClick={onToggleSidebar}
        >
          <Menu size={20} />
        </button>

        <div className="text-lg font-semibold text-[var(--text-primary)]">Welcome Back</div>
      </div>

      <div className="flex items-center gap-6">
        <nav className="hidden md:flex gap-6 text-[var(--text-secondary)]">
          <Link href="/dashboard" className="hover:text-[var(--accent-primary)] transition">Dashboard</Link>
          <Link href="/articles" className="hover:text-[var(--accent-primary)] transition">Articles</Link>
          <Link href="/wordbank" className="hover:text-[var(--accent-primary)] transition">Word Bank</Link>
        </nav>

        <button
          className="p-2 text-[var(--text-secondary)] hover:text-[var(--accent-primary)] transition"
          onClick={onToggleTheme}
        >
          {currentTheme === "light" ? <Moon size={20} /> : <Sun size={20} />}
        </button>

        {/* Profile Dropdown */}
        <div className="relative" ref={dropdownRef}>
          <button
            onClick={() => setProfileOpen(!profileOpen)}
            className="w-10 h-10 rounded-full bg-[var(--accent-primary)] flex items-center justify-center text-white font-semibold shadow-lg hover:bg-[var(--accent-tertiary)] transition"
          >
            {user?.email?.charAt(0).toUpperCase() || "U"}
          </button>

          {profileOpen && (
            <div className="absolute right-0 mt-2 w-64 bg-[var(--bg-secondary)] border border-[var(--border-primary)] rounded-xl shadow-xl overflow-hidden z-50 animate-slide-in">
              <div className="p-4 border-b border-[var(--border-primary)]">
                <div className="font-semibold text-[var(--text-primary)] truncate">
                  {user?.email}
                </div>
                <div className="text-sm text-[var(--text-secondary)]">
                  {user?.user_metadata?.name || "Language Learner"}
                </div>
              </div>

              <div className="p-2">
                <Link
                  href="/profile"
                  onClick={() => setProfileOpen(false)}
                  className="flex items-center gap-3 p-3 rounded-lg hover:bg-[var(--bg-tertiary)] transition text-[var(--text-primary)]"
                >
                  <User size={18} />
                  <span>My Profile</span>
                </Link>

                <button
                  onClick={handleLogout}
                  className="w-full flex items-center gap-3 p-3 rounded-lg hover:bg-[var(--bg-tertiary)] transition text-[var(--accent-danger)]"
                >
                  <LogOut size={18} />
                  <span>Logout</span>
                </button>
              </div>
            </div>
          )}
        </div>
      </div>
    </header>
  );
}

"use client";

import { useEffect, useState } from "react";
import { User, Globe, Award, Settings as SettingsIcon } from "lucide-react";
import { supabase } from "@/lib/supabase";

const languages = [
  "English", "Spanish", "French", "German", "Italian", "Portuguese",
  "Chinese", "Japanese", "Korean", "Arabic", "Russian", "Hindi"
];

const levels = ["A1", "A2", "B1", "B2", "C1", "C2"];

const interests = [
  "Technology", "Science", "History", "Arts", "Sports",
  "Politics", "Business", "Health", "Travel", "Food"
];

export default function ProfilePage() {
  const [profile, setProfile] = useState<any>(null);
  const [editing, setEditing] = useState(false);
  const [formData, setFormData] = useState({
    native_language: "",
    target_language: "",
    proficiency_level: "",
    interests: [] as string[],
  });

  useEffect(() => {
    loadProfile();
  }, []);

  async function loadProfile() {
    const res = await fetch("/api/profile/get");
    const data = await res.json();
    setProfile(data.profile);
    setFormData({
      native_language: data.profile?.native_language || "",
      target_language: data.profile?.target_language || "",
      proficiency_level: data.profile?.proficiency_level || "",
      interests: data.profile?.interests || [],
    });
  }

  async function handleSave() {
    const res = await fetch("/api/profile/update", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify(formData),
    });

    if (res.ok) {
      await loadProfile();
      setEditing(false);
    }
  }

  const handleInterestToggle = (interest: string) => {
    setFormData(prev => ({
      ...prev,
      interests: prev.interests.includes(interest)
        ? prev.interests.filter(i => i !== interest)
        : [...prev.interests, interest]
    }));
  };

  if (!profile) {
    return (
      <div className="flex items-center justify-center h-full">
        <div className="text-[var(--text-secondary)]">Loading profile...</div>
      </div>
    );
  }

  return (
    <div className="max-w-4xl mx-auto space-y-6">
      <div className="flex items-center justify-between">
        <div>
          <h1 className="text-3xl font-bold text-[var(--text-primary)] mb-2">Profile</h1>
          <p className="text-[var(--text-secondary)]">Manage your learning preferences</p>
        </div>
        <button
          onClick={() => editing ? handleSave() : setEditing(true)}
          className="bg-[var(--accent-primary)] hover:bg-[var(--accent-tertiary)] text-white px-6 py-3 rounded-lg font-semibold transition flex items-center gap-2"
        >
          <SettingsIcon size={20} />
          {editing ? "Save Changes" : "Edit Profile"}
        </button>
      </div>

      {/* Profile Card */}
      <div className="bg-[var(--bg-secondary)] border border-[var(--border-primary)] rounded-xl p-8 shadow-lg">
        <div className="flex items-center gap-6 mb-8">
          <div className="w-24 h-24 rounded-full bg-[var(--accent-primary)] flex items-center justify-center text-white text-4xl font-bold">
            {profile.email?.charAt(0).toUpperCase() || "U"}
          </div>
          <div>
            <h2 className="text-2xl font-bold text-[var(--text-primary)] mb-1">
              {profile.email}
            </h2>
            <p className="text-[var(--text-secondary)]">
              Member since {new Date(profile.created_at).toLocaleDateString()}
            </p>
          </div>
        </div>

        {/* Learning Details */}
        <div className="space-y-6">
          <div className="grid md:grid-cols-2 gap-6">
            {/* Native Language */}
            <div>
              <label className="flex items-center gap-2 text-[var(--text-primary)] font-semibold mb-3">
                <Globe size={20} />
                Native Language
              </label>
              {editing ? (
                <select
                  className="w-full bg-[var(--bg-tertiary)] border border-[var(--border-primary)] text-[var(--text-primary)] p-3 rounded-lg focus:border-[var(--accent-primary)] focus:outline-none"
                  value={formData.native_language}
                  onChange={e => setFormData({...formData, native_language: e.target.value})}
                >
                  <option value="">Select language</option>
                  {languages.map(lang => (
                    <option key={lang} value={lang}>{lang}</option>
                  ))}
                </select>
              ) : (
                <div className="p-3 bg-[var(--bg-tertiary)] border border-[var(--border-primary)] rounded-lg text-[var(--text-primary)]">
                  {profile.native_language || "Not set"}
                </div>
              )}
            </div>

            {/* Target Language */}
            <div>
              <label className="flex items-center gap-2 text-[var(--text-primary)] font-semibold mb-3">
                <Globe size={20} />
                Target Language
              </label>
              {editing ? (
                <select
                  className="w-full bg-[var(--bg-tertiary)] border border-[var(--border-primary)] text-[var(--text-primary)] p-3 rounded-lg focus:border-[var(--accent-primary)] focus:outline-none"
                  value={formData.target_language}
                  onChange={e => setFormData({...formData, target_language: e.target.value})}
                >
                  <option value="">Select language</option>
                  {languages.map(lang => (
                    <option key={lang} value={lang}>{lang}</option>
                  ))}
                </select>
              ) : (
                <div className="p-3 bg-[var(--bg-tertiary)] border border-[var(--border-primary)] rounded-lg text-[var(--text-primary)]">
                  {profile.target_language || "Not set"}
                </div>
              )}
            </div>
          </div>

          {/* Proficiency Level */}
          <div>
            <label className="flex items-center gap-2 text-[var(--text-primary)] font-semibold mb-3">
              <Award size={20} />
              Proficiency Level
            </label>
            {editing ? (
              <div className="flex gap-2 flex-wrap">
                {levels.map(level => (
                  <button
                    key={level}
                    onClick={() => setFormData({...formData, proficiency_level: level})}
                    className={`px-4 py-2 rounded-lg border-2 transition ${
                      formData.proficiency_level === level
                        ? "border-[var(--accent-primary)] bg-[var(--accent-primary)]/10 text-[var(--accent-primary)]"
                        : "border-[var(--border-primary)] text-[var(--text-secondary)] hover:border-[var(--accent-primary)]/50"
                    }`}
                  >
                    {level}
                  </button>
                ))}
              </div>
            ) : (
              <div className="p-3 bg-[var(--bg-tertiary)] border border-[var(--border-primary)] rounded-lg text-[var(--text-primary)] inline-block">
                {profile.proficiency_level || "Not set"}
              </div>
            )}
          </div>

          {/* Interests */}
          <div>
            <label className="flex items-center gap-2 text-[var(--text-primary)] font-semibold mb-3">
              <User size={20} />
              Interests
            </label>
            {editing ? (
              <div className="grid grid-cols-2 md:grid-cols-3 gap-3">
                {interests.map(interest => (
                  <button
                    key={interest}
                    onClick={() => handleInterestToggle(interest)}
                    className={`p-3 rounded-lg border-2 transition text-sm ${
                      formData.interests.includes(interest)
                        ? "border-[var(--accent-primary)] bg-[var(--accent-primary)]/10 text-[var(--accent-primary)]"
                        : "border-[var(--border-primary)] text-[var(--text-secondary)] hover:border-[var(--accent-primary)]/50"
                    }`}
                  >
                    {interest}
                  </button>
                ))}
              </div>
            ) : (
              <div className="flex gap-2 flex-wrap">
                {(profile.interests || []).map((interest: string) => (
                  <span
                    key={interest}

export default function ArticlePage({ params }: { params: { id: string } }) {

  const [selectedWord, setSelectedWord] = useState<string | null>(null);
  const [wordData, setWordData] = useState<any>(null);
  const [loadingWord, setLoadingWord] = useState(false);
  const [userWords, setUserWords] = useState<Record<string, string>>({});
  const articleRef = useRef<HTMLDivElement>(null);

  useEffect(() => {
    loadArticle();
    loadUserWords();
  }, [params.id]);

  async function loadArticle() {
    const res = await fetch(`/api/article/get?id=${params.id}`);
    const data = await res.json();
    setArticle(data);
  }

  async function loadUserWords() {
    const res = await fetch("/api/words/list");
    const data = await res.json();
    const wordsMap: Record<string, string> = {};
    (data.words || []).forEach((w: any) => {
      wordsMap[w.word.toLowerCase()] = w.status;
    });
    setUserWords(wordsMap);
  }

  async function handleWordClick(word: string) {
    const cleanWord = word.toLowerCase().replace(/[^a-z]/g, "");
    if (!cleanWord || cleanWord.length < 2) return;

    setSelectedWord(cleanWord);
    setLoadingWord(true);

    // Track click
    await fetch("/api/words/track", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ word: cleanWord, article_id: params.id }),
    });

    // Get definition
    const res = await fetch(`/api/words/define?word=${cleanWord}`);
    const data = await res.json();
    setWordData(data);
    setLoadingWord(false);
  }

  async function handleUpdateWordStatus(status: string) {
    if (!selectedWord) return;

    await fetch("/api/words/update", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({
        word: selectedWord,
        status,
        definition: wordData?.definition,
        difficulty: wordData?.difficulty,
        context: article?.title,
      }),
    });

    setUserWords(prev => ({ ...prev, [selectedWord]: status }));
    setSelectedWord(null);
  }

  if (!article) {
    return (
      <div className="flex items-center justify-center h-full">
        <div className="text-[var(--text-secondary)]">Loading article...</div>
      </div>
    );
  }

  const renderArticleText = () => {
    const words = article.cached_text.split(/(\s+)/);
    
    return words.map((word: string, index: number) => {
      if (!word.trim()) return <span key={index}>{word}</span>;
      
      const cleanWord = word.toLowerCase().replace(/[^a-z]/g, "");
      const status = userWords[cleanWord];
      
      return (
        <span
          key={index}
          className={`word-clickable ${
            status === "known" ? "word-known" :
            status === "learning" ? "word-learning" :
            "word-unknown"
          }`}
          onClick={() => handleWordClick(word)}
        >
          {word}
        </span>
      );
    });
  };

  return (
    <div className="max-w-4xl mx-auto space-y-6">
      {/* Header */}
      <div className="flex items-center justify-between">
        <button
          onClick={() => router.back()}
          className="flex items-center gap-2 text-[var(--text-secondary)] hover:text-[var(--accent-primary)] transition"
        >
          <ArrowLeft size={20} />
          Back
        </button>

        <div className="flex items-center gap-4">

import { NextResponse } from "next/server";
import { supabase } from "@/lib/supabase";

export async function GET(req: Request) {
  try {
    const { data: { user }, error: authError } = await supabase.auth.getUser();
    
    if (authError || !user) {
      return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
    }

    const { data: profile, error } = await supabase
      .from("profiles")
      .select("*")
      .eq("id", user.id)
      .single();

    if (error) {
      return NextResponse.json({ error: error.message }, { status: 500 });
    }

    return NextResponse.json({ profile });
  } catch (error) {
    return NextResponse.json({ error: "Server error" }, { status: 500 });
  }
}

import { NextResponse } from "next/server";
import axios from "axios";

export async function GET(req: Request) {
  try {
    const { searchParams } = new URL(req.url);
    const query = searchParams.get("q");

    if (!query) {
      return NextResponse.json({ error: "Query required" }, { status: 400 });
    }

    const response = await axios.get(
      `https://en.wikipedia.org/w/api.php`,
      {
        params: {
          action: "opensearch",
          search: query,
          limit: 10,
          namespace: 0,
          format: "json",
        },
      }
    );

    const [_, titles, descriptions, urls] = response.data;
    
    const results = titles.map((title: string, index: number) => ({
      title,
      description: descriptions[index],
      url: urls[index],
    }));

    return NextResponse.json({ results });
  } catch (error) {
    return NextResponse.json({ error: "Search failed" }, { status: 500 });
  }
}

import { NextResponse } from "next/server";
import { supabase } from "@/lib/supabase";
import axios from "axios";

export async function POST(req: Request) {
  try {
    const { data: { user }, error: authError } = await supabase.auth.getUser();
    
    if (authError || !user) {
      return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
    }

    const { title } = await req.json();

    // Fetch article summary
    const summaryRes = await axios.get(
      `https://en.wikipedia.org/api/rest_v1/page/summary/${encodeURIComponent(title)}`
    );

    // Fetch full text
    const textRes = await axios.get(
      `https://en.wikipedia.org/api/rest_v1/page/plain/${encodeURIComponent(title)}`
    );

    const text = textRes.data;
    const wordCount = text.split(/\s+/).length;

    // Calculate difficulty based on word count and complexity
    let difficulty = "B1";
    if (wordCount > 2000) difficulty = "B2";
    if (wordCount > 3500) difficulty = "C1";

    const article = {
      title: summaryRes.data.title,
      summary: summaryRes.data.extract,
      cached_text: text,
      url: summaryRes.data.content_urls.desktop.page,
      topic: summaryRes.data.title,
      word_count: wordCount,
      difficulty,
    };

    // Insert into database
    const { data: inserted, error } = await supabase
      .from("articles")
      .insert(article)
      .select()
      .single();

    if (error) {
      return NextResponse.json({ error: error.message }, { status: 500 });
    }

    // Create user article relationship
    await supabase.from("user_articles").insert({
      user_id: user.id,
      article_id: inserted.id,
      progress: 0,
      completed: false,
      started_at: new Date().toISOString(),
    });

    return NextResponse.json({ article: inserted });
  } catch (error) {
    return NextResponse.json({ error: "Failed to fetch article" }, { status: 500 });
  }
}

import { NextResponse } from "next/server";
import axios from "axios";

export async function GET(req: Request) {
  try {
    const { searchParams } = new URL(req.url);
    const word = searchParams.get("word");

    if (!word) {
      return NextResponse.json({ error: "Word required" }, { status: 400 });
    }

    // Fetch from Free Dictionary API
    const response = await axios.get(
      `https://api.dictionaryapi.dev/api/v2/entries/en/${word}`
    );

    const entry = response.data[0];
    const definition = entry.meanings[0]?.definitions[0]?.definition || "No definition found";
    const example = entry.meanings[0]?.definitions[0]?.example || "";
    const partOfSpeech = entry.meanings[0]?.partOfSpeech || "";

    // Simple difficulty estimation based on word length and syllables
    let difficulty = "B1";
    if (word.length > 8) difficulty = "B2";
    if (word.length > 12) difficulty = "C1";

    return NextResponse.json({
      word,
      definition,
      example,
      partOfSpeech,
      difficulty,
    });
  } catch (error) {
    return NextResponse.json({
      word: req.url.split("word=")[1],
      definition: "Definition not available",
      difficulty: "B1",
    });
  }
}

import { NextResponse } from "next/server";
import { supabase } from "@/lib/supabase";

export async function POST(req: Request) {
  try {
    const { data: { user }, error: authError } = await supabase.auth.getUser();
    
    if (authError || !user) {
      return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
    }

    const { word, status, definition, difficulty, context } = await req.json();

    // Check if word exists
    const { data: existing } = await supabase
      .from("user_words")
      .select("*")
      .eq("user_id", user.id)
      .eq("word", word.toLowerCase())
      .single();

    if (existing) {
      // Update existing word
      const { data, error } = await supabase
        .from("user_words")
        .update({
          status,
          last_seen: new Date().toISOString(),
          times_seen: existing.times_seen + 1,
        })
        .eq("id", existing.id)
        .select()
        .single();

      if (error) {
        return NextResponse.json({ error: error.message }, { status: 500 });
      }

      return NextResponse.json({ word: data });
    } else {
      // Insert new word
      const { data, error } = await supabase
        .from("user_words")
        .insert({
          user_id: user.id,
          word: word.toLowerCase(),
          status,
          definition: definition || "",
          difficulty: difficulty || "B1",
          context: context || "",
          last_seen: new Date().toISOString(),
          times_seen: 1,
        })
        .select()
        .single();

      if (error) {
        return NextResponse.json({ error: error.message }, { status: 500 });
      }

      return NextResponse.json({ word: data });
    }
  } catch (error) {
    return NextResponse.json({ error: "Server error" }, { status: 500 });
  }
}

export async function GET(req: Request) {

    if (authError || !user) {
      return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
    }

    const { searchParams } = new URL(req.url);
    const status = searchParams.get("status");

    let query = supabase
      .from("user_words")
      .select("*")
      .eq("user_id", user.id)
      .order("last_seen", { ascending: false });

    if (status && status !== "all") {
      query = query.eq("status", status);
    }

    const { data, error } = await query;

    if (error) {
      return NextResponse.json({ error: error.message }, { status: 500 });
    }

    return NextResponse.json({ words: data });
  } catch (error) {
    return NextResponse.json({ error: "Server error" }, { status: 500 });
  }
}
